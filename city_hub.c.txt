#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>

#define MAX_CMD_LEN 1024
#define MAX_ARGS 64

void execute_start_monitor() {
    int main_pipe[2];
    if (pipe(main_pipe) == -1) {
        return;
    }

    pid_t hub_mon_pid = fork();
    if (hub_mon_pid == -1) {
        close(main_pipe[0]);
        close(main_pipe[1]);
        return;
    }

    if (hub_mon_pid == 0) {
        close(main_pipe[0]);
        
        int mon_pipe[2];
        if (pipe(mon_pipe) == -1) {
            exit(1);
        }

        pid_t mon_pid = fork();
        if (mon_pid == -1) {
            exit(1);
        }

        if (mon_pid == 0) {
            close(mon_pipe[0]);
            close(main_pipe[1]);
            dup2(mon_pipe[1], STDOUT_FILENO);
            close(mon_pipe[1]);

            execlp("./monitor_reports", "monitor_reports", NULL);
            exit(1);
        }

        close(mon_pipe[1]);
        
        char buffer[512];
        ssize_t bytes_read;
        while ((bytes_read = read(mon_pipe[0], buffer, sizeof(buffer) - 1)) > 0) {
            buffer[bytes_read] = '\0';
            write(main_pipe[1], buffer, bytes_read);
            
            if (strstr(buffer, "deja ruleaza") != NULL || strstr(buffer, "oprit") != NULL) {
                break;
            }
        }
        
        close(mon_pipe[0]);
        close(main_pipe[1]);
        waitpid(mon_pid, NULL, 0);
        exit(0);
    }

    close(main_pipe[1]);
    
    pid_t reader_pid = fork();
    if (reader_pid == 0) {
        char buffer[512];
        ssize_t bytes_read;
        while ((bytes_read = read(main_pipe[0], buffer, sizeof(buffer) - 1)) > 0) {
            buffer[bytes_read] = '\0';
            printf("%s", buffer);
            fflush(stdout);
            if (strstr(buffer, "deja ruleaza") != NULL || strstr(buffer, "oprit") != NULL) {
                printf("[HUB] Monitorul s-a oprit în mod neașteptat sau a eșuat la pornire.\n");
                fflush(stdout);
            }
        }
        close(main_pipe[0]);
        exit(0);
    }
    
    close(main_pipe[0]);
}

void execute_calculate_scores(char *districts[], int count) {
    for (int i = 0; i < count; i++) {
        int pipefd[2];
        if (pipe(pipefd) == -1) {
            continue;
        }

        pid_t pid = fork();
        if (pid == -1) {
            close(pipefd[0]);
            close(pipefd[1]);
            continue;
        }

        if (pid == 0) {
            close(pipefd[0]);
            dup2(pipefd[1], STDOUT_FILENO);
            close(pipefd[1]);

            execlp("./scorer", "scorer", districts[i], NULL);
            exit(1);
        }

        close(pipefd[1]);

        char buffer[512];
        ssize_t bytes_read;
        printf("--- Scoruri pentru districtul: %s ---\n", districts[i]);
        while ((bytes_read = read(pipefd[0], buffer, sizeof(buffer) - 1)) > 0) {
            buffer[bytes_read] = '\0';
            printf("%s", buffer);
        }
        close(pipefd[0]);
        waitpid(pid, NULL, 0);
    }
}

int main() {
    char input[MAX_CMD_LEN];
    char *args[MAX_ARGS];

    while (1) {
        printf("city_hub> ");
        fflush(stdout);

        if (fgets(input, sizeof(input), stdin) == NULL) {
            break;
        }

        input[strcspn(input, "\n")] = '\0';

        int arg_count = 0;
        char *token = strtok(input, " ");
        while (token != NULL && arg_count < MAX_ARGS) {
            args[arg_count++] = token;
            token = strtok(NULL, " ");
        }

        if (arg_count == 0) {
            continue;
        }

        if (strcmp(args[0], "exit") == 0) {
            break;
        } 
        else if (strcmp(args[0], "start_monitor") == 0) {
            execute_start_monitor();
        } 
        else if (strcmp(args[0], "calculate_scores") == 0) {
            if (arg_count >= 2) {
                execute_calculate_scores(&args[1], arg_count - 1);
            }
        }
    }

    return 0;
}
